permutations = (list, i) ->
  switch list.length - i
    when 0
      []
    when 1
      [[list[i]]]
    when 2
      [
        [list[i], list[i + 1]]
        [list[i + 1], list[i]]
      ]
    else
      p = []
      offset = list.length - i
      for j in [0...offset]
        l = list.slice()
        swap(l, i, i + j)
        p.push(expand(l[i], permutations(l, i + 1)))
      concat.apply(null, p)
