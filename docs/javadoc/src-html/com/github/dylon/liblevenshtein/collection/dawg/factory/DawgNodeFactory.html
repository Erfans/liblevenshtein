001package com.github.dylon.liblevenshtein.collection.dawg.factory;
002
003import java.util.Queue;
004import java.util.ArrayDeque;
005
006import it.unimi.dsi.fastutil.chars.Char2ObjectMap;
007import it.unimi.dsi.fastutil.chars.Char2ObjectRBTreeMap;
008
009import com.github.dylon.liblevenshtein.collection.dawg.DawgNode;
010
011/**
012 * Builds nodes for use in DAWG structures. This implementation uses an object
013 * pool to avoid unnecessary memory consumption and garbage collection.<br>
014 * <br>
015 * <b>WARNING:</b> This implementation is not thread-safe.
016 *
017 * @author Dylon Edwards
018 * @since 2.1.0
019 */
020public class DawgNodeFactory implements IDawgNodeFactory<DawgNode> {
021
022  /**
023   * Caches {@link DawgNode} instances to recycle from {@link #build()}.
024   */
025  private final Queue<DawgNode> queue = new ArrayDeque<>();
026
027  /**
028   * {@inheritDoc}
029   */
030  @Override
031  public DawgNode build() {
032    DawgNode node = queue.poll();
033    if (null == node) {
034      final Char2ObjectMap<DawgNode> edges = new Char2ObjectRBTreeMap<>();
035      node = new DawgNode(edges);
036    }
037    return node;
038  }
039
040  /**
041   * {@inheritDoc}
042   */
043  @Override
044  public DawgNodeFactory recycle(final DawgNode node) {
045    node.clear();
046    queue.offer(node);
047    return this;
048  }
049}
