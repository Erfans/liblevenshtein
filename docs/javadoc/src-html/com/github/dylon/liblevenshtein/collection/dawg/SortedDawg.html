123    // Add the suffix, starting from the correct node, mid-way through the graph

126      final char label = term.charAt(i);
127      final DawgNode nextNode = factory.build();
128      uncheckedTransitions.addFirst(transitionFactory.build(node, label, nextNode));
129      node = nextNode;
130      i += 1;
131    }
132    finalNodes.add(node);
133    previousTerm = term;
134    size += 1;
135    return true;

147   * @param lowerBound Number of pending transitions to leave for the next
148   * round (they will be the most-recent transitions).
149   */
150  private void minimize(final int lowerBound) {
151    // Proceed from the leaf up to a certain point
152    for (int j = uncheckedTransitions.size(); j > lowerBound; --j) {
153      final Transition<DawgNode> transition = uncheckedTransitions.removeFirst();
154      final DawgNode source = transition.source();
155      final char label = transition.label();
156      final DawgNode target = transition.target();

163      } else {
164        source.addEdge(label, target);
165        minimizedNodes.put(targetKey, target);
166      }
