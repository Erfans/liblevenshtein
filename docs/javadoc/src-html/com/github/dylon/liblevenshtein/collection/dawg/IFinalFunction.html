001package com.github.dylon.liblevenshtein.collection.dawg;
002
003/**
004 * Specifies the necessary methods for functors that return whether nodes of
005 * some type represent the final states of an automaton.
006 * @param <State> Kind of the state this function accepts.
007 * @author Dylon Edwards
008 * @since 2.1.0
009 */
010public interface IFinalFunction<State> {
011
012  /**
013   * Whether the state represents the last character of some term.
014   * @param current State to check for finality.
015   * @return Whether this node represents the last character of some term.
016   */
017  boolean at(State current);
018}
