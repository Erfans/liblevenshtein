001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * Transitions a standard, Levenshtein position to all possible positions, given
005 * a set of parameters.
006 * @author Dylon Edwards
007 * @since 2.1.0
008 */
009public class StandardPositionTransitionFunction
010  extends AbstractPositionTransitionFunction {
011
012  /**
013   * {@inheritDoc}
014   */
015  @Override
016  public IState of(
017      final int n,
018      final int[] position,
019      final boolean[] characteristicVector,
020      final int offset) {
021
022    final int i = position[0];
023    final int e = position[1];
024    final int h = i - offset;
025    final int w = characteristicVector.length;
026
027    if (e < n) {
028      if (h <= w - 2) {
029        final int a = (n - e < Integer.MAX_VALUE)
030          ? n - e + 1
031          : Integer.MAX_VALUE;
032        final int b = w - h;
033        final int k = (a < b) ? a : b;
034        final int j = indexOf(characteristicVector, k, h);
035
036        if (0 == j) {
037          return stateFactory.build(
038              positionFactory.build((1 + i), e));
039        }
040
041        if (j > 0) {
042          return stateFactory.build(
043              positionFactory.build(i, (e + 1)),
044              positionFactory.build((i + 1), (e + 1)),
045              positionFactory.build((i + j + 1), (e + j)));
046        }
047
048        // else, j < 0
049        return stateFactory.build(
050            positionFactory.build(i, (e + 1)),
051            positionFactory.build((i + 1), (e + 1)));
052      }
053
054      if (h == w - 1) {
055        if (characteristicVector[h]) {
056          return stateFactory.build(
057              positionFactory.build((i + 1), e));
058        }
059
060        return stateFactory.build(
061            positionFactory.build(i, (e + 1)),
062            positionFactory.build((i + 1), (e + 1)));
063      }
064
065      // else, h == w
066      return stateFactory.build(
067          positionFactory.build(i, (e + 1)));
068    }
069
070    if ((e == n) && (h <= w - 1) && characteristicVector[h]) {
071      return stateFactory.build(
072          positionFactory.build((i + 1), n));
073    }
074
075    return null;
076  }
077}
