001package com.github.dylon.liblevenshtein.levenshtein.factory;
002
003import com.github.dylon.liblevenshtein.levenshtein.Element;
004
005/**
006 * Builds elements that are used to link position vectors in Levenshtein states.
007 * @param <Type> Kind of position vectors that will be linked.
008 * @author Dylon Edwards
009 * @since 2.1.0
010 */
011public interface IElementFactory<Type> {
012
013  /**
014   * Builds a new linked-list node with the given value.
015   * @param value Value to assign the linked-list node.
016   * @return New linked-list node with the specified value.
017   */
018  Element<Type> build(Type value);
019
020  /**
021   * Recycles an {@link Element} for re-use in {@link #build(Object) build}.
022   * @param element {@link Element} to recycle.  Once you've recycled an element
023   * you should discard its reference.
024   */
025  default void recycle(Element<Type> element) {
026        // default behavior is to do nothing
027  }
028}
