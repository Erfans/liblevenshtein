001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * Defines operations for merging states together.
005 * @author Dylon Edwards
006 * @since 2.1.0
007 */
008public interface IMergeFunction {
009
010  /**
011   * Merges the positions in the latter state into the former, in a
012   * subsumption-friendly manner.  You should assume that once this method has
013   * returned, some of the positions in the latter state may have been recycled
014   * and it should therefore be recycled, too (DO NOT RE-USE IT !!!).
015   *
016   * @param state The state into which the positions of the latter should be
017   * merged.
018   *
019   * @param positions The state from which the positions should be merged.
020   */
021  void into(IState state, IState positions);
022}
