001package com.github.dylon.liblevenshtein.levenshtein.factory;
002
003import com.github.dylon.liblevenshtein.levenshtein.IStateTransitionFunction;
004
005/**
006 * Builds functions that transition states to other states, given some evidence.
007 * @author Dylon Edwards
008 * @since 2.1.0
009 */
010public interface IStateTransitionFactory {
011
012  /**
013   * Builds a new state-transition function that only considers spelling
014   * candidates no more than {@code maxDistance} errors from the query term.
015   * @param maxDistance Maximum number of errors tolerated in transforming the
016   * query term into the spelling candidate.
017   * @return New state-transition function that only considers spelling
018   * candidates within {@code maxDistance} errors from the query term.
019   */
020  IStateTransitionFunction build(int maxDistance);
021
022  /**
023   * Recycles a state-transition function so it may be re-used in
024   * {@link #build(int) build}.
025   * @param transition {@link IStateTransitionFunction} to recycle.  You must
026   * discard your reference to {@code transition} once you've recycled it.
027   */
028  default void recycle(IStateTransitionFunction transition) {
029    // default behavior is to do nothing
030  }
031}
