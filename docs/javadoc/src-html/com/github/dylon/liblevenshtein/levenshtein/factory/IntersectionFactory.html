001package com.github.dylon.liblevenshtein.levenshtein.factory;
002
003import java.util.Queue;
004import java.util.ArrayDeque;
005
006import com.github.dylon.liblevenshtein.levenshtein.Intersection;
007import com.github.dylon.liblevenshtein.levenshtein.IState;
008
009/**
010 * Builds intersections between a dictionary automaton and a Levenshtein
011 * automaton.
012 * @param <DictionaryNode> Kind of nodes in the dictionary automaton.
013 * @author Dylon Edwards
014 * @since 2.1.0
015 */
016public class IntersectionFactory<DictionaryNode>
017  implements IIntersectionFactory<DictionaryNode> {
018
019  /**
020   * Object pool for recycled {@link Intersection}s, to be returned from
021   * {@link #build(String,Object,IState) build}.
022   */
023  private final Queue<Intersection<DictionaryNode>>
024    intersections = new ArrayDeque<>();
025
026  /**
027   * {@inheritDoc}
028   */
029  @Override
030  public Intersection<DictionaryNode> build(
031      final String candidate,
032      final DictionaryNode dictionaryNode,
033      final IState levenshteinState) {
034
035    Intersection<DictionaryNode> intersection = intersections.poll();
036
037    if (null == intersection) {
038      intersection = new Intersection<DictionaryNode>();
039    }
040
041    intersection.candidate(candidate);
042    intersection.dictionaryNode(dictionaryNode);
043    intersection.levenshteinState(levenshteinState);
044    return intersection;
045  }
046
047  /**
048   * {@inheritDoc}
049   */
050  @Override
051  public void recycle(final Intersection<DictionaryNode> intersection) {
052    intersection.candidate(null);
053    intersection.dictionaryNode(null);
054    intersection.levenshteinState(null);
055    intersections.offer(intersection);
056  }
057}
