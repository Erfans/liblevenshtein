001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * Specifies the interface that all distance functions must implement.
005 * @author Dylon Edwards
006 * @since 2.1.0
007 */
008public interface IDistance<Term> {
009
010  /**
011   * Finds the distance between two terms, {@code v} and {@code w}. The distance
012   * between two terms is complemented by their similarity, which is determined
013   * by subtracting their distance from the maximum distance they may be apart.
014   * @param v Term to compare with {@code w}
015   * @param w Term to compare with {@code v}
016   * @return Distance between {@code v} and {@code w}
017   */
018  int between(Term v, Term w);
019}
